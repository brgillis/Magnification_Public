90  flt_type _mftau( const flt_type & tau, const flt_type & conc ) const // Mtot/Mvir from tau
91  {
92  if(tau<=0) return 0;
93  flt_type M0oM200 = 1 / ( std::log( 1 + conc ) - conc / ( 1 + conc ) );
94  flt_type tautau = tau*tau;
95  flt_type result =
96  M0oM200 * tautau / square( tautau + 1 )
97  * ( ( tautau - 1 ) * std::log( tau ) + tau * pi
98  - ( tautau + 1 ) );
99  return result;
100  }
