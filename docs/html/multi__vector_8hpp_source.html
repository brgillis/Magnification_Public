522  // Reshaping
523  void reshape(const shape_t & new_shape, const T & new_val = T())
524  {
525  // Check if no reshaping is necessary
526  if(_shape_==new_shape) return;
527 
528  if(new_shape.size() > std::numeric_limits<dsize_t>::max())
529  throw std::out_of_range("Cannot reshape vector: Dimensions array too large.");
530 
531  vsize_t new_size = 1;
532  for(shape_t::const_iterator itr=new_shape.begin(); itr!=new_shape.end();++itr)
533  new_size *= *itr;
534 
535  if(new_size >= vec::max_size())
536  throw std::out_of_range("Cannot reshape vector: Too large to be indexed by this implementation.");
537  // There's a slim chance this will throw on a valid size, but unlikely
538 
539  vec::resize(new_size,new_val);
540  _shape_ = new_shape;
541  _num_dim_ = _shape_.size();
542  }
