512  extra_shear_factor*sigma_crit(_host_ptr_->z(),2*_host_ptr_->z());
513  }
514  else
515  {
516  return _host_ptr_->Delta_Sigma(R_actual)*angle_factor +
517  extra_shear_factor*sigma_crit(_host_ptr_->z(),2*_host_ptr_->z()); //TODO Should there be a factor of 1/2 here?
