187 template<typename T1, typename A1=std::allocator<T1>, typename T2=T1, typename A2=std::allocator<T2>>
188 T2 interpolate_bins(const T2 & val, const std::vector<T1,A1> & lim_vec, const std::vector<T2,A2> & val_vec)
189 {
190  assert(ssize(val_vec)>1);
191  assert(ssize(lim_vec)==ssize(val_vec)+1);
192  assert(is_monotonically_increasing(lim_vec));
193  assert(ssize(lim_vec)>2);
194 
195  ssize_t bin_i=ssize(lim_vec)-3;
196 
197  for(ssize_t i=0; i<ssize(lim_vec)-2; ++i)
198  {
199  if((lim_vec[i]+lim_vec[i+1])/2>=val)
200  {
201  if(i==0)
202  bin_i=0;
203  else
204  bin_i=i-1;
205  break;
206  }
207  }
208 
209  T1 xlo = (lim_vec[bin_i]+lim_vec[bin_i+1])/2;
210  T1 xhi = (lim_vec[bin_i+1]+lim_vec[bin_i+2])/2;
211  const T2 & ylo = val_vec[bin_i];
212  const T2 & yhi = val_vec[bin_i+1];
213 
214  return ylo + (yhi-ylo)/(xhi-xlo) * (val-xlo);
215 }
