63  const char major_tag = Eigen::ColMajor,
64  const char precision = 10)

66  // Set up the value we'll print if an entry is bad
67  std::stringstream ss("");
68  ss << std::numeric_limits<flt_type>::max();
69  std::string bad_value = ss.str();
70  ss.clear();
71  ss.str("");
72 
73  // Set the precision of this stringstream
74  ss.precision(precision);
75 
76  size_t num_col;
77  size_t num_row;
78 
79  if(major_tag==Eigen::ColMajor)
80  {
81  num_col = ssize(data);

83  }

88  }

92  const bool skip_header = (ssize(header)==0);
93 
94  try
95  {
96  bool add_comment_marker = false;
97 
98  // First, we loop through to get the maximum width of each column
99  // Check the header first
100  if(!skip_header)
101  {
102  for ( size_t c = 0; c < num_col; c++ )
103  {
104  if(c==0)
105  {
106  // For the first header element, we'll check if it starts with a
107  // '#'. If it doesn't, we'll add one to it.
108  ss.str(header[c]);
109  if ( ss.peek() != (int_type)( *"#" ) )

112  }
113  }

117  }
118  } // for( int_type c = 0; c < num_columns; c++ )
119  }
120 
121  // Now loop through the data
122  for ( size_t i = 0; i < num_row; i++ )
123  {
124  for ( size_t c = 0; c < num_col; c++ )
125  {
126  T val;
127  if(major_tag==Eigen::ColMajor)

147  if ( str_val.length() > width[c] )
148  {

151  } // for( int_type c = 0; c < num_columns; c++ )
152  } // for( int_type i = 0; i < num_rows; i++ ) (testing width)

159  }

163  for ( size_t c = 1; c < num_col; c++ )

171  {
172  out_stream << "# ";

182  {
183  if(add_comment_marker)
184  {
185  out_stream << " ";
186  }
187  for ( size_t c = 0; c < num_col; c++ )

283 // Load a table's header as a vector of strings
284 inline header_t load_header( std::istream & table_stream )
285 {
286  std::string temp_line;
287  std::vector<header_t> possible_headers;
288 
289  // Get all comment lines at the top of the file
290  while ( table_stream )
291  {
292  if ( table_stream.peek() == (int_type)( *"#" ) )
293  {
294  getline( table_stream, temp_line );
295 
296  header_t temp_header = convert_to_header(temp_line);
297 
298  if(ssize(temp_header) > 0)
299  possible_headers.push_back(temp_header);
300  }
301  else
302  {
303  break;
304  }
305  }
306 
307  if(ssize(possible_headers)==1) return possible_headers[0];
308  if(ssize(possible_headers)==0) return header_t();
309 
310  // If we get here, more than one line is a possible header candidate. Our next step is to
311  // go to the data and count the columns in the first line. If only one possible header has
312  // the right length, we know that's the one.
313 
314  int_type n_cols = 0;
315  do
316  {
317  getline( table_stream, temp_line );
318  std::string junk;
319  std::istringstream line_data_stream(temp_line);
320  while (line_data_stream >> junk)
321  {
322  ++n_cols;
323  }
324  } while(n_cols==0 && table_stream);
325 
326  if(n_cols==0) // If we can't find any data
327  {
328  return header_t();
329  }
330 
331  // Search through the possible headers, and see if we find exactly one with the right size
332  int_type num_right_size = 0;
333  ssize_t i_best = 0;
334  for(ssize_t i=0; i<ssize(possible_headers); ++i)
335  {
336  if(ssize(possible_headers[i])==n_cols)
337  {
338  ++num_right_size;
339  i_best = i;
340  }
341  }
342 
343  if(num_right_size != 1) // If multiple or zero lines are the right size
344  {
345  return header_t();
346  }
347 
348  return possible_headers[i_best];
349 }
