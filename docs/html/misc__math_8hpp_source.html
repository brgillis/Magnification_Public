196 // Rounds to nearest integer, favouring even
197 inline int_type round_int( flt_type value, const flt_type & epsilon=std::numeric_limits<flt_type>::epsilon() )
198 {
199 
200  if ( value < 0.0 )
201  return -round_int( -value, epsilon );
202 
203  flt_type ipart;
204  std::modf( value, &ipart );
205 
206  // If 'value' is exactly halfway between two integers
207  if ( fabs( value - ( ipart + 0.5 ) ) < epsilon )
208  {
209  // If 'ipart' is even then return 'ipart'
210  if ( std::fmod( ipart, 2.0 ) < epsilon )
211  return (int_type)ipart;
212 
213  // Else return the nearest even integer
214  return (int_type)std::ceil( ipart + 0.5 );
215  }
216 
217  // Otherwise use the usual round to closest
218  // (Either symmetric half-up or half-down will do0
219  return (int_type)std::floor( value + 0.5 );
220 }

353 // Distance between two vectors, where the dimensions are weighted by vector c
354 template< typename Ta, typename Tb,
355 BRG_F_NOT_CONTAINER(Ta), BRG_F_NOT_CONTAINER(Tb) >
356 inline Ta weighted_dist( const std::vector< Ta > & a,
357  const std::vector< Tb > & b )
358 {
359  Ta result = 0;
360  assert(a.size()==b.size());
361  for ( size_t i = 0; i < a.size(); i++ )
362  {
363  result += square( b[i] - a[i] );
364  }
365  result = sqrt( result );
366  return result;
367 }

369 BRG_F_NOT_CONTAINER(Ta), BRG_F_NOT_CONTAINER(Tb), BRG_F_NOT_CONTAINER(Tc) >
370 inline Ta weighted_dist( const std::vector< Ta > & a,
371  const std::vector< Tb > & b, const std::vector< Tc > & c )
372 {
