46 // Generate a case-resampled version of a vector or array
47 template< typename T_vec, typename T_gen=decltype(rng) >
48 T_vec get_case_resampled(const T_vec & vec, T_gen & gen=rng)
49 {
50  T_vec res(vec);
51 
52  std::uniform_int_distribution<int_type> dist(0,vec.size()-1);
53 
54  for( auto & v : res )
55  {
56  v = vec[dist(gen)];
57  }
58 
59  return res;
60 }

62 // Get a jackknife resample - note that this doesn't preserve ordering
63 template< typename T_vec,
64  std::enable_if<IceBRG::is_eigen_container<T_vec>::value,char> = 0>
65 T_vec get_jackknife_resampled(const T_vec & vec,
66  const int_type & sample_num = 0,
67  const int_type & num_samples = 2)
68 {
69  assert(num_samples>=1);
70  assert(sample_num>=0);
71  assert(sample_num<num_samples);
72 
73  T_vec res(vec);
74 
75  res.conservativeResize(std::partition(
76  res.data(),
77  res.data()+res.size(),
78  [&vec](const int_type & i){return vec(i) % num_samples == sample_num;})-res.data());
79 
80  return res;
81 }
