94  // _new_key(...) - Generates a new key of the templated type, which is not already in the map
95  template< typename new_key_type, typename other_map_type=insertion_ordered_map<key_type,mapped_type>,
96  typename std::enable_if<std::is_arithmetic<new_key_type>::value,new_key_type>::type* = nullptr,
97  typename std::enable_if<std::is_convertible<key_type,typename other_map_type::key_type>::value,new_key_type>::type* = nullptr>
98  new_key_type _new_key(const other_map_type & other_map = other_map_type())
99  {
100  // For arithmetic types, we'll start by setting the new key value equal to the number of
101  // elements in the map. If that's in, we'll add 1. Repeat till we find an unused value.
102  // If we try more values than are in the key map, something's gone wrong, so throw an
103  // exception.
104  new_key_type test_key = ssize(_key_map_);
105  size_type counter=0;
106  while((_key_map_.find(test_key) != _key_map_.end()) &&
107  (other_map.find(test_key) != other_map.end()) &&
108  (counter<=ssize(_key_map_)))
109  {
110  test_key += 1;
111  ++counter;
112  }
113  if(counter>ssize(_key_map_)) throw std::runtime_error("Cannot generate new arithmetic key for insertion_ordered_map.");
114  return test_key;
115  }

117  template< typename new_key_type, typename other_map_type=insertion_ordered_map<key_type,mapped_type>,
118  typename std::enable_if<!std::is_arithmetic<new_key_type>::value,new_key_type>::type* = nullptr,
119  typename std::enable_if<std::is_convertible<key_type,typename other_map_type::key_type>::value,new_key_type>::type* = nullptr>
120  new_key_type _new_key(const other_map_type & other_map = other_map_type())
121  {
122  // For strings, we'll start by setting the new key value to "col_N", where N is the number of
123  // elements in the map. If that's in, we'll add 1 to N. Repeat till we find an unused value.
124  // If we try more values than are in the key map, something's gone wrong, so throw an
125  // exception.
126  size_type N = ssize(_key_map_);
127 
128  std::stringstream ss;
129 
130  ss.str();
131  ss << "col_" << N;
132  std::string test_key = ss.str();
133 
134  size_type counter=0;
135  while((_key_map_.count(test_key) + other_map.count(test_key) > 0) &&
136  (counter<=ssize(_key_map_)+other_map.size()))
137  {
138  ss.str();
139  ss << "col_" << ++N;
140  test_key = ss.str();
141 
142  ++counter;
143  }
144  if(counter>ssize(_key_map_)+other_map.size()) throw std::runtime_error("Cannot generate new string key for insertion_ordered_map.");
145  return test_key;
146  }

420  template< typename new_key_type >
421  char change_key(const key_type & init_key, new_key_type && new_key)
422  {
423  // Get the position of the value we're going to be altering
424  auto it = _key_map_.find(init_key);
425 
426  if(it==_key_map_.end()) return 1;
427  if(_key_map_.count(new_key)>0) return 2;
428 
429  size_t pos = it->second;
430 
431  // Alter the value in the key map by erasing old entry and adding new
432  _key_map_.erase(it);
433  _key_map_[std::forward<new_key_type>(new_key)] = pos;
434 
435  // Alter the value in the vector
436  _val_vector_[pos].first = std::forward<new_key_type>(new_key);
437 
438  return 0;
