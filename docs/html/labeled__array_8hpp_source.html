267  void _add_column_buffer_to_data_table() const
268  {
269  // Check that we actually do have columns to add
270  if(_column_buffer_.empty()) return;
271 
272  // Get the number of columns
273  const size_type num_cols_to_add = ssize(_column_buffer_);
274  const size_type old_num_cols = _data_table_.cols();
275  const size_type new_num_cols = _data_table_.cols() + num_cols_to_add;
276  size_type num_rows;
277 
278  // If the data table is initially empty, we'll get the number of rows from the first
279  // buffered column
280  if(ssize(_data_table_)==0)
281  {
282  num_rows = ssize(_column_buffer_.begin()->second);
283  }
284  else // Otherwise use the current number of rows
285  {
286  num_rows = _data_table_.rows();
287  }
288 
289  // Check that all columns we'll be adding are the proper size before we change anything
290 
291  for( const auto & column : _column_buffer_)
292  {
293  // Check that the column is the proper size
294  if(ssize(column.second)!=num_rows)
295  {
296  throw std::logic_error("All columns added to a labeled_array must have the same size.\n");
297  }
298  }
299 
300  // Resize the data table, conserving existing elements
301  _data_table_.conservativeResize(num_rows, new_num_cols);
302 
303  // Go through the buffer and add each column to the data table. If the label already exists,
304  // replace that column instead
305  size_type current_column = old_num_cols;
306  size_type num_preexisting_cols = 0; // number of labels which we find already exist
307  for( const auto & column : _column_buffer_)
308  {
309  const_label_type & label = column.first;
310  auto & column_data = column.second;
311 
312  // Check if the label already exists
313  if(_label_map_.left.count(label)==1)
314  {
315  // Instead of adding a new column, we'll replace the existing one
316  const size_type & column_index = _label_map_.left.at(label);
317  for(size_type i=0; i<ssize(column_data); ++i)
318  {
319  _data_table_(i,column_index) = column_data[i];
320  }
321  ++num_preexisting_cols;
322  }
323  else // New column to add
324  {
325  // Add this to the label map
326  _label_map_.left.insert(std::make_pair(label,current_column));
327  for(size_type i=0; i<ssize(column_data); ++i)
328  {
329  _data_table_(i,current_column) = column_data[i];
330  }
331 
332  // Increment the current column index
333  ++current_column;
334  }
335  }
336 
337  // If we overwrote any preexisting columns, resize the data table now to the proper size
338  if(num_preexisting_cols>0)
339  {
340  _data_table_.conservativeResize(Eigen::NoChange_t(), new_num_cols-num_preexisting_cols);
341  }
342 
343  // Clear the buffer, and we're done
344  _clear_column_buffer();
345 
346  }

348  void _add_row_buffer_to_data_table() const
349  {
350  // Check that we actually do have rows to add
351  if(_row_buffer_.empty()) return;
352 
353  // Get the number of rows
354  const size_type num_rows_to_add = ssize(_row_buffer_);
355  const size_type old_num_rows = _data_table_.rows();
356  const size_type new_num_rows = _data_table_.rows() + num_rows_to_add;
357  size_type num_cols;
358 
359  // If the data table is initially empty, we'll get the number of columns from the first
360  // buffered row
361  if(ssize(_data_table_)==0)
362  {
363  num_cols = ssize(_row_buffer_.front());
364  }
365  else // Otherwise use the current number of columns
366  {
367  num_cols = _data_table_.cols();
368  }
369 
370  // Check that all rows we'll be adding are the proper size before we change anything
371 
372  for( const auto & row : _row_buffer_)
373  {
374  // Check that the row is the proper size
375  if(static_cast<size_type>(ssize(row))!=num_cols)
376  {
377  throw std::logic_error("All rows added to a labeled_array must have the same size.\n");
378  }
379  }
380 
381  // If this is the first row being added, check if we need to create a dummy label map
382  if(ssize(_data_table_)==0)
383  {
384  if(static_cast<size_type>(ssize(_label_map_.left))!=num_cols)
385  {
386  _generate_dummy_label_map(num_cols);
387  assert(static_cast<size_type>(ssize(_label_map_.left))==num_cols);
388  }
389  }
390 
391  // Resize the data table, conserving existing elements
392  _data_table_.conservativeResize(new_num_rows, num_cols);
393 
394  size_type current_row = old_num_rows;
395  // Go through the buffer and add each row to the data table
396  for( const auto & row : _row_buffer_)
397  {
398  for(typename row_buffer_row_type::size_type i=0; i<row.size(); ++i)
