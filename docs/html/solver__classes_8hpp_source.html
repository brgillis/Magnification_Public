76  const array_type & solve( const f & func, const array_type & init_in_params,
77  const array_type & init_min_in_params,

80  const size_type & max_steps=1000000,
81  const size_type & annealing_period=100000,
82  const flt_type & annealing_factor=4,
83  const size_type & skip_factor=10)
84  {
85  static_assert(std::is_convertible<decltype(func(array_type())),value_type>::value,
86  "Function passed to MCMC solver has incorrect return type.");
87 
88  assert(skip_factor>0);
89  assert(max_steps>=2*annealing_period);
90 
91  assert(init_in_params.size()==init_min_in_params.size());
92  assert(init_in_params.size()==init_max_in_params.size());
93  assert(init_in_params.size()==init_in_param_step_sigmas.size());
94  assert((init_in_params.size()==T_num_params) or (T_num_params==Eigen::Dynamic));
95 
96  size_type array_size = init_in_params.size();
97 
98  clear();
99 
100  bool bounds_check = true;
101 
102  // Check how bounds were passed in
103  if((init_min_in_params.size()==0) && (init_max_in_params.size()==0))
104  bounds_check = false;
105 
106  array_type min_in_params = init_min_in_params;
107  array_type max_in_params = init_max_in_params;
108 
109  if(min_in_params.size() < max_in_params.size())
110  {
111  min_in_params.resize(max_in_params.size(),std::numeric_limits<value_type>::lowest());
112  }
113  if(max_in_params.size() < min_in_params.size())
114  {
115  max_in_params.resize(min_in_params.size(),std::numeric_limits<value_type>::max());
116  }
117 
118  // Check if any of the params likely have max and min mixed up
119  if(bounds_check)
120  {
121  for(int_type i = 0; i < max_in_params.size(); i++)
122  {
123  if(min_in_params(i)>max_in_params(i))
124  {
125  std::swap(min_in_params(i),max_in_params(i));
126  } // if(min_in_params.at(i)>max_in_params.at(i))
127  } // for(int_type i = 0; i < max_in_params.size(); i++)
128  } // if(bounds_check)
129 
130  // Check step sizes
131 
132  array_type in_param_step_sigmas = init_in_param_step_sigmas;
133 
134  if(in_param_step_sigmas.size() < min_in_params.size())
135  in_param_step_sigmas.resize(min_in_params.size(),0);
136 
137  for(int_type i = 0; i < in_param_step_sigmas.size(); i++)
138  {
139  if(in_param_step_sigmas(i) <= 0)
140  {
141  if(bounds_check)
142  {
143  in_param_step_sigmas(i) =
144  (max_in_params(i)-min_in_params(i))/10.;
145  }
146  else
147  {
148  in_param_step_sigmas(i) = 1; // Default behaviour
149  }
150  }
151  }
152 
153  // Initialise generally
154  flt_type annealing = 1.;
155 
156  bool last_cycle = false;
157  size_type last_cycle_count = 0;
158 
159  array_type current_in_params = init_in_params, test_in_params = init_in_params,
160  best_in_params = init_in_params;
161  value_type out_param, best_out_param;
162  array_type mean_in_params = array_type::Zero(min_in_params.size());
163 
164  // Set up second cycle data
165  bool first_cycle = true;
166  bool second_cycle = false;
167  size_type second_cycle_points_to_record = annealing_period/skip_factor;
168  _test_run_points_ = array2d_type(array_size,second_cycle_points_to_record);
169 
170  // Get value at initial point
171  try
172  {
173  out_param = func(test_in_params);
174  }
175  catch(const std::exception &e)
176  {
177  std::cerr << "Cannot execute solve_MCMC at initial point.\n";
178  throw;
179  }
180  best_out_param = out_param;
181  flt_type last_log_likelihood = -annealing*out_param/2;
182 
183  auto new_Gaus_rand = [] (const value_type & dummy) {return IceBRG::Gaus_rand();};
184 
185  for(int_type step = 0; step < max_steps; step++)
186  {
187  // Get a new value
188  test_in_params = current_in_params + array_type::Zero(array_size).unaryExpr(new_Gaus_rand)*in_param_step_sigmas/annealing;
189 
190  // Check if it's in bounds
191  if(bounds_check)
192  {
193  test_in_params = bound(min_in_params,test_in_params,max_in_params);
194  }
195 
196  // Find the result for this value
197  bool good_result = true;
198  try
199  {
200  out_param = func(test_in_params);
201  }
202  catch(const std::exception &e)
203  {
204  good_result = false;
205  }
206  flt_type new_log_likelihood = -annealing*out_param/2;
207 
208  // If it's usable, check if we should step to it
209  bool step_to_it = false;
210  if(good_result)
211  {
212  if(new_log_likelihood>=last_log_likelihood)
213  {
214  step_to_it = true;
215  }
216  else
217  {
218  if(drand() < std::exp(new_log_likelihood - last_log_likelihood))
219  step_to_it = true;
220  }
221  if(step_to_it)
222  {
223  last_log_likelihood = new_log_likelihood;
224  current_in_params = test_in_params;
225 
226  // Check if we have a new best point
227  if(out_param < best_out_param)
228  {
229  best_out_param = out_param;
230  best_in_params = current_in_params;
231  }
232  }
233  }
234 
235  // If we're on the second cycle, add to the recorded list of test points
236  if(second_cycle)
237  {
238  if(IceBRG::divisible(step,skip_factor))
239  {
240  size_type c_i = (step-annealing_period)/skip_factor-1;
241  auto col = _test_run_points_->col(c_i);
242  for(size_type p_i=0; p_i<array_size; ++p_i)
243  {
244  col(p_i) = current_in_params(p_i);
245  }
246  }
247  }
248 
249  // If we're on the last cycle, add to the mean
250  if(last_cycle)
251  {
252  mean_in_params += current_in_params;
253  last_cycle_count += 1;
254  }
255 
256  if((step!=0)&&(IceBRG::divisible(step,annealing_period)))
257  {
258  if(first_cycle) // Special handling for end of first cycle
259  {
260  first_cycle = false;
261  second_cycle = true;
262  }
263  else
264  {
265  second_cycle = false;
266  annealing *= annealing_factor;
267 
268  // Recalculate likelihood
269  last_log_likelihood *= annealing_factor;
270 
271  // Check if we're going into the last cycle
272  if(max_steps-step<=annealing_period)
273  last_cycle = true;
274  }
275  }
276  } // for(int_type step = 0; step < max_steps; step++)
277 
278  // Calculate mean now
279  mean_in_params = mean_in_params/IceBRG::safe_d(last_cycle_count);
280 
281  // Check if mean actually gives a better best
282  try
283  {
284  out_param = func(mean_in_params);
285  if(out_param < best_out_param)
286  {
287  best_in_params = mean_in_params;
288  best_out_param = out_param;
289  }
290  }
291  catch(const std::exception &e)
292  {
293  // Just leave it, no need to do anything
294  }
295 
296  _best_fits_ = std::move(best_in_params);
297 
298  return *_best_fits_;
299  }
