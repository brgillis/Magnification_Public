39 std::vector<T,A> make_linear_limit_vector_base(const T & min, const T & max, const size_t & num_bins)
40 {
41  if(max<=min)
42  throw std::logic_error("max must be greater than min for limit vector.\n");
43  if(num_bins<1)
44  throw std::logic_error("num_bins must be greater than zero for limit vector.\n");
45  if(( isinf(min) || isinf(max) ) && (num_bins>1))
46  throw std::logic_error("Cannot generate limit vector with and infinite limits and multiple bins.");
47 
48  T step = (max-min)/(flt_type)num_bins;
49 
50  std::vector<T,A> result(1,min);
51 
52  if(!isinf(step))

57  result.push_back( min + step*(flt_type)limit_num );
58  }
