43 // "Safe" functions - perform the operation specified, but will
44 // take necessary actions to ensure it won't crash the program
45 // if the argument is invalid (eg. taking square-root of a negative
46 // number).
47 template< class T >
48 const T safe_sqrt( const T & a )

50 
51 #ifdef _BRG_WARN_FOR_SAFE_FUNCTIONS_TRIGGERED_
